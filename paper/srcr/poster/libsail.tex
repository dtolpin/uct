\newcommand{\boat}[1]{ 
	\begin{scope}[shift={#1}]
		\begin{scope}[scale=1.5]
			\draw [very thick,black] (3,1) -- (3,7.25);
            \draw [thick,black,fill=red!70!white] (3,7)
                  .. controls +(1.5,0) and +(1,0) .. (5,1)
			      .. controls +(0.25,0.5) and +(1, 0.3) .. (3,1)
			      .. controls +(1, 0) and +(1,0) .. (3,7);
			\draw [thick,black,fill=red!70!white] (3,5.5) 
                  .. controls +(-0.3, -1) and +(0,0) .. (1,1)
                  .. controls +(0.5,0.1) and +(-0.5,0.1) .. (2.5,1)
                  .. controls +(-0.25,1) and +(0,0) .. (3,5.5);
			\draw [very thick,black,fill=lightgray] 
				(1,0) -- (5,0) -- (6,1.5)
				.. controls +(-2,-1) and +(0,0) .. (0,1) -- cycle;
		\end{scope}
	\end{scope}}

\newcommand{\lake}[2] {
	\begin{scope}[shift={#2}]
		\begin{scope}[scale={#1}]
			\draw[draw=black!70!blue,very thick] (0,0) rectangle (10,10);
		    \draw[fill=red!40!white] (9,9) rectangle +(1,1);
			\foreach \x in {1,...,9} {
			    \draw[draw=black!70!blue] (0,\x) -- (10,\x);
			    \draw[draw=black!70!blue] (\x,0) -- (\x,10);}
		\end{scope}
	\end{scope}}

\newcommand{\location}[2]{
	\lake {5} {#1};
        \draw [gray,dashed,very thick,->] {#2} +(3,3)
         .. controls +(-5,30) and +(5,-30) .. (47,47);
	\boat {#2};}
